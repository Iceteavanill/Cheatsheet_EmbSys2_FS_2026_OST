\subsection{Memory Types}
% GOAL V05 • You are familiar with the basic types of embedded memory.
  \vspace{-0.1cm}
  \begin{center}
    \begin{tabular}{llll}
      \toprule
      \textbf{Type} & \textbf{Volatile} & \textbf{Writable}   & \textbf{Typical Use} \\ \midrule
      SRAM          & yes               & yes                 & Cache, Stack, Heap, Variables \\
      DRAM          & yes               & yes                 & Main memory (typ.\ external) \\
      ROM           & no                & no                  & Boot code, Constants \\
      Flash         & no                & yes (block/sector erase) & Program code, Persistent data \\
      EEPROM        & no                & yes (byte-wise write)    & Configuration, Calibration data \\
      \bottomrule
    \end{tabular}
  \end{center}
  \vspace{-0.1cm}
  \textit{SRAM is fast, requires no refresh; DRAM is denser, cheaper, and requires periodic refresh.}
