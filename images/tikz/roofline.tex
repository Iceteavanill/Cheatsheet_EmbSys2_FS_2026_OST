\begin{tikzpicture}[>=Latex, thick, font=\Huge, xscale=1.35, yscale=1.05]

  % ---------- Coordinate system note ----------
  % Both axes are on a log2 scale. A coordinate (x,y) in this picture
  % corresponds to I(n) = 2^x [flops/byte] and P(n) = 2^y [flops/cycle].
  % Here we set pi = 2 and beta = 1, so pi/beta = 2 (matches the figure).
  % The axes meet in the lower-left corner (xmin,ymin), not at I=1,P=1.

  \def\xmin{-2.6}
  \def\xmax{3.9}
  \def\ymin{-2.6}
  \def\ymax{3.3}

  % ---------- Shaded regions (draw first, behind everything) ----------
  % Pink: compute-bound region (right of the dotted line, up to the pi bound)
  \fill[red!12] (1,\ymin) rectangle (\xmax,1);
  % Blue: memory-bound region (triangle under the beta*I line, left of dotted line)
  \fill[blue!12] (\xmin+0.3,\ymin) -- (1,\ymin) -- (1,1) -- cycle;

  % ---------- Axes (meeting in the lower-left corner) ----------
  \draw[->] (\xmin,\ymin) -- (\xmax,\ymin) node[right] {$I(n)$ in flops/byte};
  \draw[->] (\xmin,\ymin) -- (\xmin,\ymax) node[above] {$P(n)$ in flops/cycle};

  % ---------- x tick marks and labels ----------
  \foreach \x/\lbl in {-2/{1/4}, -1/{1/2}, 0/1, 1/2, 2/4, 3/8} {
    \draw (\x,\ymin+0.06) -- (\x,\ymin-0.06);
    \node[below] at (\x,\ymin-0.08) {$\lbl$};
  }

  % ---------- y tick marks and labels ----------
  \foreach \y/\lbl in {-2/{1/4}, -1/{1/2}, 0/1, 1/2, 2/4} {
    \draw (\xmin+0.06,\y) -- (\xmin-0.06,\y);
    \node[left] at (\xmin-0.08,\y) {$\lbl$};
  }

  % ---------- Bound lines ----------
  % Diagonal line: bound based on beta,  P <= beta*I  (slope 1, passes through I=1,P=1)
  \draw[ultra thick] (\xmin+0.3,\ymin) -- (2.35,2.35);
  \node[above right] at (2.35,2.35) {bound based on $\beta$: $P \le \beta I$};

  % Horizontal line: bound based on pi,  P <= pi
  \draw[ultra thick] (\xmin,1) -- (\xmax,1);
  \node[above right] at (1.8,1) {bound based on $\pi$: $P \le \pi$};

  % ---------- Dotted vertical line at the intersection ----------
  \draw[densely dotted] (1,\ymin) -- (1,1);

  % ---------- Sample program point ----------
  \node at (2,0.35) {\textbf{x}};
  \draw[->, line width=0.5mm] (2.9,-0.55) -- (2.15,0.3);
  \node[right] at (2.9,-0.35) {some program};
  \node[right] at (2.9,-0.75) {run on some input};

  % ---------- Region labels ----------
  \node[blue!70!black] at (-1.3,\ymin-1) {memory bound};
  \node[red!70!black] at (3,\ymin-1) {compute bound};

  % ---------- pi/beta annotation ----------
  \node at (0.5,\ymin-1.1) {$\pi/\beta$};
  \draw[->, line width=0.5mm] (0.55,\ymin-0.9) -- (0.95,\ymin-0.08);

  % ---------- New bound (red) ----------
  \draw[ultra thick, olive!70] (\xmin,2) -- (\xmax,2);
  \node[above right, olive!70] at (-2,2) {new bound: $4\pi$};
  \draw[densely dotted, olive!70] (3,\ymin) -- (3,3);

  % ---------- New bound annotation ----------
  \draw[->, olive!70, line width=1mm] (1,\ymin+1) -- (3,\ymin+1);
  \node[olive!70, align=left, anchor=west] at (-4,\ymin-1.8) {If $\beta$ does not change more programs become memory bound};
